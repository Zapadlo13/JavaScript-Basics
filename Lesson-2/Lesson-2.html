1. Дан код:

var a = 1, b = 1, c, d;
c = ++a; alert(c); // 2 к а применина префиксная форма возвращает новое значение
d = b++; alert(d); // 1 к b применина постфиксная форма возвращает старое значение
c = (2+ ++a); alert(c); // 5 а уже 2 раза применина префиксная форма 2 +(1+1+1)
d = (2+ b++); alert(d); // 4 b будет 3 тольок после примера ответ 4
alert(a); // 3 2 раза была добавлена +1
alert(b); // 3 2 раза была добавлена +1
